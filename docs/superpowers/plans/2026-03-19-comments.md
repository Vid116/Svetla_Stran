# Comments Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add public comments with threading, anonymous posting, and editor moderation to article pages.

**Architecture:** All visitor comments go to `pending` status (no AI moderation for now). Editor comments auto-approve. Editors moderate inline on the article page. One level of reply nesting. Client component fetches via API routes, server uses `getSupabaseAdmin()`.

**Tech Stack:** Next.js App Router, Supabase PostgreSQL, Tailwind CSS, TypeScript

**Simplification from spec:** No AI moderation, no auto-replies, no rate limiting. All visitor comments → `pending`. These will be added when the AI pipeline moves to production hosting.

```sql
CREATE TABLE comments (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  article_id       uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  parent_id        uuid REFERENCES comments(id) ON DELETE CASCADE,
  author_name      text NOT NULL CHECK (char_length(author_name) <= 50),
  author_type      text NOT NULL DEFAULT 'visitor',
  editor_id        uuid REFERENCES editors(id),
  body             text NOT NULL CHECK (char_length(body) <= 2000),
  status           text NOT NULL DEFAULT 'pending',
  rejection_reason text,
  created_at       timestamptz NOT NULL DEFAULT now(),
  updated_at       timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX idx_comments_article_status ON comments (article_id, status);
CREATE INDEX idx_comments_parent ON comments (parent_id);
```

```typescript
export async function getCommentsByArticle(articleId: string, includeAll = false) {
  const supabase = getSupabaseAdmin();
  let query = supabase
    .from("comments")
    .select("*")
    .eq("article_id", articleId)
    .order("created_at", { ascending: true });

  if (!includeAll) {
    query = query.eq("status", "approved");
  }

  const { data, error } = await query;
  if (error) throw error;
  return data || [];
}
```

```typescript
export async function moderateComment(id: string, status: string, rejectionReason?: string) {
  const supabase = getSupabaseAdmin();
  const updates: Record<string, any> = {
    status,
    updated_at: new Date().toISOString(),
  };
  if (rejectionReason) updates.rejection_reason = rejectionReason;

  const { error } = await supabase.from("comments").update(updates).eq("id", id);
  if (error) throw error;

  // If rejecting a parent comment, also reject its children
  if (status === "rejected") {
    await supabase
      .from("comments")
      .update({ status: "rejected", updated_at: new Date().toISOString() })
      .eq("parent_id", id);
  }
}
```

- [ ] **Step 4: Add deleteComment function**

```typescript
export async function deleteComment(id: string) {
  const supabase = getSupabaseAdmin();
  const { error } = await supabase.from("comments").delete().eq("id", id);
  if (error) throw error;
}
```

Create `app/api/comments/route.ts`:

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getAuthEditor } from "@/lib/require-auth-api";
import {
  getCommentsByArticle,
  createComment,
  moderateComment,
  deleteComment,
} from "@/lib/db";

export async function GET(req: NextRequest) {
  try {
    const articleId = req.nextUrl.searchParams.get("articleId");
    if (!articleId) {
      return NextResponse.json({ error: "articleId required" }, { status: 400 });
    }

    const editor = await getAuthEditor();
    const comments = await getCommentsByArticle(articleId, !!editor);

    return NextResponse.json({ comments, isEditor: !!editor, editorName: editor?.name || null });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

- [ ] **Step 2: Add POST endpoint**

```typescript
export async function POST(req: NextRequest) {
  try {
    const { articleId, parentId, authorName, body } = await req.json();

    // Validate
    if (!articleId || !authorName?.trim() || !body?.trim()) {
      return NextResponse.json({ error: "Missing required fields" }, { status: 400 });
    }
    if (authorName.length > 50) {
      return NextResponse.json({ error: "Ime predolgo (max 50 znakov)" }, { status: 400 });
    }
    if (body.length > 2000) {
      return NextResponse.json({ error: "Komentar predolg (max 2000 znakov)" }, { status: 400 });
    }

    const editor = await getAuthEditor();

    const comment = await createComment({
      article_id: articleId,
      parent_id: parentId || null,
      author_name: editor ? editor.name : authorName.trim(),
      author_type: editor ? "editor" : "visitor",
      editor_id: editor?.id || null,
      body: body.trim(),
      status: editor ? "approved" : "pending",
    });

    return NextResponse.json(comment);
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

```typescript
export async function PUT(req: NextRequest) {
  const denied = await (await import("@/lib/require-auth-api")).requireAuthAPI();
  if (denied) return denied;

  try {
    const { commentId, action, body, rejectionReason } = await req.json();

    if (!commentId || !action) {
      return NextResponse.json({ error: "Missing commentId or action" }, { status: 400 });
    }

    if (action === "approve") {
      await moderateComment(commentId, "approved");
    } else if (action === "reject") {
      await moderateComment(commentId, "rejected", rejectionReason);
    } else if (action === "reply") {
      if (!body?.trim()) {
        return NextResponse.json({ error: "Reply body required" }, { status: 400 });
      }
      const editor = await getAuthEditor();
      await createComment({
        article_id: (await import("@/lib/db")).getSupabaseAdmin()
          ? "" : "", // Will be set below
        parent_id: commentId,
        author_name: editor!.name,
        author_type: "editor",
        editor_id: editor!.id,
        body: body.trim(),
        status: "approved",
      });
    }

    return NextResponse.json({ ok: true });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

```typescript
export async function PUT(req: NextRequest) {
  const { requireAuthAPI } = await import("@/lib/require-auth-api");
  const denied = await requireAuthAPI();
  if (denied) return denied;

  try {
    const { commentId, action, body, rejectionReason } = await req.json();

    if (!commentId || !action) {
      return NextResponse.json({ error: "Missing commentId or action" }, { status: 400 });
    }

    const editor = await getAuthEditor();

    if (action === "approve") {
      await moderateComment(commentId, "approved");
    } else if (action === "reject") {
      await moderateComment(commentId, "rejected", rejectionReason);
    } else if (action === "reply") {
      if (!body?.trim()) {
        return NextResponse.json({ error: "Reply body required" }, { status: 400 });
      }
      // Get parent comment to find article_id
      const { getSupabaseAdmin } = await import("@/lib/supabase");
      const { data: parent } = await getSupabaseAdmin()
        .from("comments")
        .select("article_id")
        .eq("id", commentId)
        .single();
      if (!parent) {
        return NextResponse.json({ error: "Comment not found" }, { status: 404 });
      }
      await createComment({
        article_id: parent.article_id,
        parent_id: commentId,
        author_name: editor!.name,
        author_type: "editor",
        editor_id: editor!.id,
        body: body.trim(),
        status: "approved",
      });
    }

    return NextResponse.json({ ok: true });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```
